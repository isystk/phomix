 11 /**
 12  * ランキング情報を取得します。
 13  * @param {Object} data
 14  * @param {function} callback
 15  */
 16 TopService.prototype.getRanking = function(data, callback) {
 17 	next(function() {
 18 		var deferred = new Deferred();
 19 		// ユーザー情報を合計ポイント順で取得します
 20 		commonDao.list('user', {where: {}, select: null, sort: {totalScore: -1},skip: data.index, limit: 20}, function(err, result) {
 21 			var rankings = [];
 22 			for (var i=0,len=result.length; i<len; i++) {
 23 				var user = result[i];
 24 				var ranking = {
 25 					snsType: user.snsType,
 26 					screenName: user.screenName || '',
 27 					name: user.name || '',
 28 					image: user.image,
 29 					totalScore: user.totalScore || 0,
 30 					highScore: user.highScore || 0,
 31 					rank: data.index + i + 1
 32 				};
 33 				rankings[i] = ranking;
 34 			}
 35 			deferred.call(rankings);
 36 		});
 37 		return deferred;
 38 	}).next(function(ranking) {
 39 		commonDao.count('user', {}, function(err, result) {
 40 			var res = {
 41 				ranking: ranking,
 42 				max: result
 43 			};
 44 			callback(err, res);
 45 		});
 46 	});
 47 };
