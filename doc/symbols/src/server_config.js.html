  7 var fs = require('fs');

  9 /**
 10  * 設定
 11  */
 12 var Config = function () {
 13 };
 14 
 15 module.exports = new Config();
 16 Config.prototype.setup = function (path, callback) {
 17 	var json = fs.readFileSync(path);
 18 	var conf = JSON.parse(json);
 19 	for (var key in conf) {
 20 		if (conf.hasOwnProperty(key)) {
 21 			this[key] = conf[key];
 22 		}
 23 	}
 24 };
