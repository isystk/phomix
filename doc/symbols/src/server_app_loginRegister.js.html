 16 /**
 17  * プロフィール画像登録
 18  * @param {Object} user
 19  * @param {Object} data
 20  */
 21 LoginRegisterControl.prototype.registProfileImage = function(user, data) {
 22 	// ファイルサイズチェック
 23 	// 画像ファイルチェック
 24 
 25 	next(function() {
 26 		var deferred = new Deferred();
 27 		// 画像のアップロード
 28 		imageService.upload(data.files, function(err, result) {
 29 			if (err) {
 30 				user.sendError({message: err.errors});
 31 				return;
 32 			}
 33 			deferred.call(result);
 34 		});
 35 		return deferred;
 36 	}).next(function(data) {
 37 		imageService.convert(data, function(err, result) {
 38 			if (err) {
 39 				user.sendError({message: err.errors});
 40 				return;
 41 			}
 42 			user.send('loginRegister.registProfileImage', {imageId: result.imageId});
 43 		});
 44 	});
 45 };

 47 /**
 48  * プロフィール画像削除
 49  * @param {Object} user
 50  * @param {Object} data
 51  */
 52 LoginRegisterControl.prototype.deleteProfileImage = function(user, data) {
 53 
 54 	// ログイン済みの場合は、自分の画像IDかどうかをチェック
 55 	// 非ログインの場合は、画像IDのユーザーがnullであることをチェック
 56 
 57 	next(function() {
 58 		imageService.remove(data.imageId, function(err, result) {
 59 			if (err) {
 60 				user.sendError({message: err.errors});
 61 				return;
 62 			}
 63 			user.send('loginRegister.deleteProfileImage', {imageId: result.imageId});
 64 		});
 65 	});
 66 };

 68 /**
 69  * プロフィール登録
 70  * @param {Object} user
 71  * @param {Object} data
 72  */
 73 LoginRegisterControl.prototype.registProfile = function(user, data) {
 74 	next(function() {
 75 		userService.regist(data, function(err, result) {
 76 			if (err) {
 77 				user.sendError({message: err.errors});
 78 				return;
 79 			}
 80 			user.send('loginRegister.registProfile', {accessToken: result.accessToken});
 81 			// ユーザーログ
 82 			userService.log({'type': LOG_TYPE.LOGIN_REGIST.key, userId: result.userId, data:{userName: result.name}});
 83 		});
 84 	});
 85 };
