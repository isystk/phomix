1 var Deferred = require('jsdeferred').Deferred;
  2 Deferred.define();
  3 
  4 var LOG_TYPE = require('../logType').LOG_TYPE;
  5 var userService = require('../service/user');
  6 
  7 /**
  8  * ログイン
  9  */
 10 function LoginControl() {
 11 }
 12 
 13 module.exports = new LoginControl();
 14 
 15 
 16 /**
 17  * ログイン処理
 18  * @param {Object} user
 19  * @param {Object} data
 20  */
 21 LoginControl.prototype.login = function(user, data) {
 22 	next(function() {
 23 		var deferred = new Deferred();
 24 		// ログインチェックを行います。
 25 		userService.login(data.email, data.password, function(err, result) {
 26 			if (err) {
 27 				user.sendError({message: err.errors});
 28 				return;
 29 			}
 30 			deferred.call(result);
 31 		});
 32 		return deferred;
 33 	}).next(function(data) {
 34 		user.send('login.login', {accessToken: data.accessToken});
 35 
 36 		// ユーザーログ
 37 		userService.log({'type': LOG_TYPE.LOGIN.key, userId: data.userId, data:{userName: data.name}});
 38 	});
 39 };
 40 
 41 /**
 42  * 自動ログイン処理
 43  * @param {Object} user
 44  * @param {Object} data
 45  */
 46 LoginControl.prototype.autoLogin = function(user, data) {
 47 
 48 	next(function() {
 49 		var deferred = new Deferred();
 50 		// ログインチェックを行います。
 51 		userService.getUserInfo(data.accessToken, function(err, result) {
 52 			if (err) {
 53 				user.sendError({message: err.errors});
 54 				return;
 55 			}
 56 			deferred.call(result);
 57 		});
 58 		return deferred;
 59 	}).next(function(data) {
 60 		if (data) {
 61 			user.send('common.autoLogin', {userId: data.userId, name: data.name, imageId: data.imageId});
 62 		} else {
 63 			user.send('common.autoLogin', {});
 64 		}
 65 
 66 	});
 67 };
