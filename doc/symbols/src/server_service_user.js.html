 15 /**
 16  * オートコンプリート用サーチを実行します
 17  *
 18  * @param data
 19  * @param callback
 20  * @return
 21  */
 22 UserService.prototype.autocomplete = function(data, callback) {
 23 
 24 	var colname = 'user';
 25 	var term = data.term ? data.term : '';
 26 	var size = data.size ? data.size : 10;
 27 	var reg = data.reg ? data.reg : '';
 28 
 29 	var field = 'name';
 30 	if (data.field) {
 31 		field = data.field;
 32 	}
 33 
 34 	if (data.reg == 'forward') { // 前方一致
 35 		reg = new RegExp('^' + term + '.*');
 36 	} else if (data.reg == 'exact') { // 完全一致
 37 		reg = term;
 38 	} else {
 39 		reg = new RegExp('.*' + term + '.*', 'i');
 40 	}
 41 
 42 	var query = {};
 43 	query[field] = reg;
 44 
 45 	var result = { list: [] };
 46 	commonDao.list(colname, {where: query, limit: size}, function(err, result) {
 47 		var list = [];
 48 		for (var i=0,len=result.length; i<len; i++) {
 49 			var item = result[i];
 50 			list.push({
 51 				label: item._id + ' (' + item.name + ')',
 52 				value: item._id
 53 			 });
 54 		}
 55 		callback(null, list);
 56 	});
 57 };

 92 /**
 93  * ユーザー情報を新規登録します。
 94  * @param {Object} data
 95  * @param {function} callback
 96  */
 97 UserService.prototype.regist = function(data, callback) {
 98 
 99 	var user = {
100 		name: data.name || '',
101 		password: data.password || '',
102 		email: data.email || '',
103 		introduction: data.introduction || '',
104 		imageId: data.imageId || null,
105 	};
106 
107 	// パスワードを暗号化する。
108 	if (user.password !== '') {
109 		user.password = passwordService.createHash(user.password);
110 	}
111 
112 	// 入力チェック
113 	var errors = [];
114 	next(function() {
115 		if (!user.name || user.name === '') {
116 			errors.push(message.ERR.USER.NAME.REQUIRE);
117 		}
118 		if (!user.password || user.password === '') {
119 			errors.push(message.ERR.USER.PASSWORD.REQUIRE);
120 		}
121 		if (!user.email || user.email === '') {
122 			errors.push(message.ERR.USER.EMAIL.REQUIRE);
123 		}
124 	}).next(function() {
125 		var deferred = new Deferred();
126 		if (user.imageId === null) {
127 			deferred.call();
128 			return;
129 		}
130 		// 画像IDの存在チェック
131 		imageService.isExist(user.imageId, function(err, result) {
132 			if(!result) {
133 				errors.push(message.ERR.USER.IMAGE_ID.INVALID);
134 			}
135 			deferred.call();
136 		});
137 		return deferred;
138 	}).next(function() {
139 		var deferred = new Deferred();
140 		if (user.email === '') {
141 			deferred.call();
142 			return;
143 		}
144 		// メールアドレスの存在チェック
145 		commonDao.list('user', {where: {email: user.email}}, function(err, result) {
146 			if(0<result.length) {
147 				errors.push(message.ERR.USER.EMAIL.USED);
148 			}
149 			deferred.call();
150 		});
151 		return deferred;
152 	}).next(function() {
153 		if (0 < errors.length) {
154 			callback({errors: errors});
155 			return;
156 		}
157 
158 		var deferred = new Deferred();
159 
160 		if(user.imageId !== '') {
161 			user.imageId = commonDao.makeObjectId(user.imageId);
162 		}
163 		user.mailChkFlg = false;
164 		user.accessToken = '';
165 		user.registTime = Date.now();
166 		user.updateTime = Date.now();
167 		user.deleteFlg = false;
168 
169 		console.log('ユーザー情報を新規登録します。', user);
170 		commonDao.upsert('user', null, user, function(err, result) {
171 			deferred.call(result);
172 		});
173 
174 		return deferred;
175 	}).next(function(data) {
176 		var deferred = new Deferred();
177 
178 		var userId = data._id;
179 		var accessToken = accessTokenService.createHash(userId);
180 
181 		commonDao.upsert('user', {_id: userId}, {accessToken: accessToken}, function(err, result) {
182 			data.accessToken = accessToken;
183 			deferred.call(data);
184 		});
185 
186 		return deferred;
187 	}).next(function(data) {
188 
189 		var userId = data._id;
190 
191 		if (user.imageId !== '') {
192 			// 画像テーブルのuserIdを更新する。
193 			commonDao.upsert('image', {_id: user.imageId}, {userId: userId}, function(err, result) {
194 				callback(err, {accessToken: data.accessToken});
195 			});
196 		} else {
197 			callback(err, {accessToken: data.accessToken, userId: userId, name: data.name});
198 		}
199 	});
200 };

202 /**
203  * ログインしてアクセストークンを取得します。
204  * @param {Object} email
205  * @param {Object} password
206  * @param {function} callback
207  */
208 UserService.prototype.login = function(email, password, callback) {
209 	// 入力チェック
210 	var errors = [];
211 	next(function() {
212 		if (!email || email === '') {
213 			errors.push(message.ERR.USER.EMAIL.REQUIRE);
214 		}
215 		if (!password || password === '') {
216 			errors.push(message.ERR.USER.PASSWORD.REQUIRE);
217 		}
218 	}).next(function() {
219 		var deferred = new Deferred();
220 		// ログインチェックを行います。
221 		commonDao.find('user', {email: email}, function(err, result) {
222 			if(!result || result.length <= 0) {
223 				errors.push(message.ERR.USER.EMAIL.NOTEXIST);
224 			}
225 			if(result && 2<=result.length) {
226 				errors.push(message.ERR.SYSTEM.INVALID, '同一のEMAILを持つユーザー情報が存在します。');
227 			}
228 			if(result && !passwordService.validateHash(result.password, password)) {
229 				errors.push(message.ERR.USER.PASSWORD.MISS);
230 			}
231 			deferred.call(result);
232 		});
233 		return deferred;
234 	}).next(function(data) {
235 
236 		if (0 < errors.length) {
237 			callback({errors: errors});
238 			return;
239 		}
240 
241 		var userId = data._id;
242 		var accessToken = accessTokenService.createHash(userId);
243 
244 		commonDao.upsert('user', {_id: userId}, {accessToken: accessToken}, function(err, result) {
245 			if (err) {
246 				callback(err, null);
247 				return;
248 			}
249 			callback(null, {
250 				userId: userId,
251 				name: data.name,
252 				accessToken: accessToken
253 			});
254 		});
255 
256 	});
257 };

259 /**
260  * アクセストークンからユーザー情報を取得します。
261  * @param {Object} accessToken
262  * @param {function} callback
263  */
264 UserService.prototype.getUserInfo = function(accessToken, callback) {
265 	// 入力チェック
266 	var errors = [];
267 	next(function() {
268 		if (!accessToken || accessToken === '') {
269 			errors.push(message.ERR.USER.ACCESS_TOKEN.REQUIRE);
270 		}
271 	}).next(function() {
272 		var deferred = new Deferred();
273 		// アクセストークンからユーザー情報を取得します。
274 		commonDao.find('user', {accessToken: accessToken}, function(err, result) {
275 			if(result && 2<=result.length) {
276 				errors.push(message.ERR.SYSTEM.INVALID, '同一のアクセストークンを持つユーザー情報が存在します。');
277 			}
278 			deferred.call(result);
279 		});
280 		return deferred;
281 	}).next(function(data) {
282 
283 		if (0 < errors.length) {
284 			callback({errors: errors});
285 			return;
286 		}
287 
288 		// アクセストークンが存在しない場合
289 		if(!data || data.length <= 0) {
290 			callback(null, {});
291 		} else {
292 
293 			var userId = data._id;
294 
295 			callback(null, {
296 				userId: userId,
297 				name: data.name,
298 				imageId: data.imageId
299 			});
300 
301 		}
302 	});
303 };
