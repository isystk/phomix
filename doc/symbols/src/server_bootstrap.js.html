  6 var path = require('path');
  7 var fs = require('fs');
  8 var args = require('./args').parse();
  9 var config = require('./config');
 10 
 11 var usage = "usage: " + process.argv[1] + '\n'
 12 	+ '\n'
 13 	+ '\t-p\tlistenポートを指定します。-p 9999\n'
 14 	+ '\t-h\tヘルプを表示します。\n'
 15 	+ '\n';
 16 
 17 /**
 18  * 起動処理
 19  * @param {function} callback
 20  */
 21 exports.setup = function(dirname, app, callback) {
 22 	// サーバー起動モードの選択
 23 	var mode = 'local';
 24 	for (var i in process.argv) {
 25 		if (i) {
 26 			var arg = process.argv[i];
 27 			if (arg === '--dev') {
 28 				mode = 'develop';
 29 			} else if (arg === '--stg') {
 30 				mode = 'staging';
 31 			} else if (arg === '--real') {
 32 				mode = 'real';
 33 			}
 34 		}
 35 	}
 36 
 37 	// ヘルプを表示
 38 	if (args['h']) {
 39 		process.stderr.write(usage);
 40 		process.exit(2);
 41 	}
 42 
 43 	// 引数チェック
 44 	check();
 45 
 46 	console.log('Configure the server mode [', mode, ']');
 47 
 48 	var configFile = dirname+'/conf/'+mode+'.json';
 49 
 50 	// config load
 51 	config.setup(configFile);
 52 
 53 	// コンフィグにサーバー起動モードを追加する
 54 	config.mode = mode;
 55 
 56 	// PIDファイル
 57 	var prefix = path.basename(process.argv[1]);
 58 	var pidfile = config.piddir+'/'+prefix+'.'+'pid';
 59 	fs.writeFileSync(pidfile, String(process.pid));
 60 
 61 	system(); // システム設定
 62 
 63 	if (callback !== undefined && callback !== null) {
 64 		callback(null);
 65 	}
 66 };
 67 
 68 /**
 69  * 引数チェック
 70  */
 71 var check = function () {
 72 	var f = false;
 73 
 74 	for (var key in args) {
 75 		if (args.hasOwnProperty(key)) {
 76 			if (args[key] === true) {
 77 				if (key === 'node' ||
 78 					key === '--local' ||
 79 					key === '--dev' ||
 80 					key === '--stg' ||
 81 					key === '--real'
 82 				) {
 83 					continue;
 84 				}
 85 				console.log(key + '" option value is not set.\n');
 86 				f = true;
 87 			}
 88 		}
 89 	}
 90 
 91 	if (f) {
 92 		process.stderr.write(usage);
 93 		process.exit(2);
 94 	}
 95 };
 96 
 97 /**
 98  * node.jsの設定を行う
 99  */
100 var system = exports.system = function(uncaughtException, sigterm, sigint, exit) {
101 	// キャッチされない例外処理を追加
102 	var _uncaughtException = function(err) {
103 		console.log(err);
104 	};
105 
106 	// kill -15
107 	var _sigterm = function() {
108 		console.log("Got SIGTERM");
109 		_exit();
110 	};
111 
112 	// Ctrl+c
113 	var _sigint = function() {
114 		console.log("Got SIGINT");
115 		_exit();
116 	};
117 
118 	// システム終了
119 	var _exit = function() {
120 		console.log("Server is going down");
121 		clean(function(){
122 			process.exit(0);
123 		});
124 	};
125 	_uncaughtException = uncaughtException || _uncaughtException;
126 	_sigterm = sigterm || _sigterm;
127 	_sigint = sigint || _sigint;
128 	_exit = exit || _exit;
129 	
130 	process.on('uncaughtException', _uncaughtException);
131 	process.on('SIGTERM', _sigterm);
132 	process.on("SIGINT", _sigint);
133 	//process.on("exit", _exit);
134 	
135 };
136 
137 var clean = function(callback) {
138 	callback(null);
139 };
