 22 /**
 23  * 管理ツールの操作ログを記録します
 24  * @param {String} name ログイン名
 25  * @param {Object} docs
 26  * @param {function} callback
 27  */
 28 AdminService.prototype.oplog = function(user, type, docs, callback) {
 29 	var now = new Date().getTime();
 30 
 31 	// IPアドレスを取得する
 32 	var getIpAddr = function(name, family) {
 33 		family = family || 'IPv4';
 34 		var ifs = os.networkInterfaces();
 35 		if (ifs.hasOwnProperty(name) === true) {
 36 			var addrs = ifs[name];
 37 			for (var i = 0; i < addrs.length; i++) {
 38 				if (family === addrs[i].family) {
 39 					return addrs[i].address;
 40 				}
 41 			}
 42 		}
 43 		return undefined;
 44 	};
 45 
 46 	// IP取得
 47 	var ip = getIpAddr(config.admin.nic);
 48 
 49 	commonDao.upsert('admin_oplog', null, {name: user.id, type:type, ip:ip, time:now, data: docs}, function(err, result) {
 50 		callback();
 51 	});
 52 };

 54 /**
 55  * 管理ログイン処理
 56  * @param data ログインデータ({name,password})
 57  * @param callback コールバック({name,session,roles})
 58  */
 59 AdminService.prototype.login = function(data, callback) {
 60 	commonDao.find('admin_user', {userId: data.userId, password: data.password}, function(err, result) {
 61 
 62 		var genSessionKey = function(value, key) {
 63 			var hash = crypto.createHash('sha256');
 64 			hash.update(value + key);
 65 			return hash.digest('base64');
 66 		};
 67 
 68 		var key = genSessionKey(result.userId, config.sessionBase);
 69 		callback(null, {
 70 			userId: result.userId,
 71 			name: result.name,
 72 			session: key
 73 		});
 74 	});
 75 };

 77 /**
 78  * セッションベースのログイン
 79  * @param data ログイデータ({session})
 80  * @param callback コールバック({name,session,roles})
 81  */
 82 AdminService.prototype.loginSession = function(data, callback) {
 83 
 84 	var userId = data.userId;
 85 	var session = data.session;
 86 
 87 	var genSessionKey = function(value, key) {
 88 		var hash = crypto.createHash('sha256');
 89 		hash.update(value + key);
 90 		return hash.digest('base64');
 91 	};
 92 
 93 	var key = genSessionKey(userId, config.sessionBase);
 94 	if (key == session) {
 95 		commonDao.find('admin_user', {userId: userId}, function(err, result) {
 96 			callback(null, {
 97 				userId: result.userId,
 98 				name: result.name,
 99 				session: key
100 			});
101 		});
102 	} else {
103 		callback(new Error(), null);
104 	}
105 };

107 /**
108  * ログインセッションキーの取得/更新
109  * @param {name, sessionBase} data
110  * @returns {string} sha256+base64されたセッションキー
111  */
112 AdminService.prototype.genSessionKey = function(data) {
113 	var hash = crypto.createHash('sha256');
114 	hash.update(data.name + config.admin.sessionBase);
115 	return hash.digest('base64');
116 };

118 /**
119  * データベース内の一覧取得
120  * @param {String} colname コレクション名
121  * @param {Object} data
122  * @param {function} callback
123  */
124 AdminService.prototype.list = function(colname, data, callback) {
125 	var page = data.page ? Math.max(1, data.page) : 1;
126 	var size = data.size ? Math.max(1, data.size) : 20;
127 	var selector = data.selector ? data.selector : {};
128 	var sort = {_id: 1};
129 
130 	if (data.sort) {
131 		sort = data.sort;
132 	}
133 
134 	next(function() {
135 		var deferred = new Deferred();
136 		commonDao.count(colname, selector, function(err, result) {
137 			deferred.call(result);
138 		});
139 		return deferred;
140 	}).next(function(count) {
141 		commonDao.list(colname, {where: selector, sort: sort, skip: (page-1) * size, limit: size}, function(err, list) {
142 			var maxpage = Math.floor(count / size);
143 			if (count % size !== 0) {
144 				maxpage = maxpage + 1;
145 			}
146 			var result = {
147 				list: list,
148 				page: page,
149 				maxpage: maxpage
150 			};
151 			callback(null, result);
152 		});
153 	});
154 };

175 /**
176  * データをダウンロード
177  * @param {String} colname コレクション名
178  * @param {Object} selector
179  * @param {Object} sort
180  * @param {Object} res
181  */
182 AdminService.prototype.download = function(colname, selector, sort, res) {
183 	var date = new Date();
184 	var zero = function(num) {
185 		var str = String(num);
186 		if (str.length == 1) {
187 			return '0' + str;
188 		} else {
189 			return str;
190 		}
191 	};
192 	var now = date.getFullYear()
193 	+ zero(date.getMonth()+1)
194 	+ zero(date.getDate())
195 	+ zero(date.getHours())
196 	+ zero(date.getMinutes())
197 	+ zero(date.getSeconds());
198 	
199 	res.writeHead(200, {'Content-Type': 'text-plain', 'Content-disposition': 'attachment;filename=hunter_' + config.mode + '_' + colname + '_' + now + '.json'});
200 
201 	commonDao.list(colname, {where: selector, sort: sort}, function(err, list) {
202 		if (err) {
203 			res.write(err.stack);
204 			res.end();
205 		}
206 		for (var i=0, len=list.length; i<len; i++) {
207 			var item = list[i];
208 			res.write(JSON.stringify(item));
209 			res.write('\n');
210 		}
211 		res.end();
212 	});
213 };

216 /**
217  * データをアップロード
218  * @param {String} colname コレクション名
219  * @param {Object} data
220  * @param {function} callback
221  */
222 AdminService.prototype.upload = function(colname, selector, clean, path, callback) {
223 
224 	next(function() {
225 		// データをクリア
226 		if (clean === 'true') {
227 			var deferred = new Deferred();
228 			commonDao.remove(colname, selector, function(err) {
229 				deferred.call();
230 			});
231 			return deferred;
232 		}
233 	}).next(function() {
234 		var deferred = new Deferred();
235 		var readFile = function () {
236 			var list = [];
237 			var stream = fs.createReadStream(path);
238 			var carry = carrier.carry(stream);
239 			carry.on('line', function(line) {
240 				var data = JSON.parse(line);
241 				var id = new commonDao.ObjectID(data._id);
242 				delete data._id;
243 				list.push({
244 					_id: id,
245 					data: data
246 				});
247 			});
248 			carry.on('end', function(end) {
249 				deferred.call(list);
250 			});
251 		}
252 		readFile();
253 		return deferred;
254 	}).next(function(list) {
255 		console.log('アップロードしたデータを登録します。', list);
256 		for (var i=0, len=list.length; i<len; i++) {
257 			var item = list[i];
258 			commonDao.upsert(colname, {_id: item._id}, item.data, function(err, result) {
259 			});
260 		}
261 		callback();
262 	});
263 };
