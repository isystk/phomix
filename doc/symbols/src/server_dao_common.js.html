1 var mongoose = require('mongoose');
  2 var config = require('../config');
  3 var colmap = {};
  4 var db = {};
  5 
  6 function CommonDao() {
  7 	db = mongoose.connect('mongodb://'+config.mongodb.hostname+'/'+config.mongodb.database);
  8 	var schemas = config.mongodb.schemas;
  9 	for (var i=0,len=schemas.length; i<len; i++) {
 10 		var schema = schemas[i];
 11 		for (var key in schema) {
 12 			var model = new mongoose.Schema(schema[key]);
 13 			mongoose.model(key, model);
 14 			colmap[key] =  mongoose.model(key);
 15 		}
 16 	}
 17 }
 18 
 19 module.exports = new CommonDao();

 21 /**
 22  * オブジェクトIDを生成します。
 23  */
 24 var makeObjectId = CommonDao.prototype.makeObjectId = function(id) {
 25 	if (id) {
 26 		return new db.mongo.ObjectID(id);
 27 	} else {
 28 		return new db.mongo.ObjectID();
 29 	}
 30 }

 32 /**
 33  * 存在チェック
 34  * @param {String} colname コレクション名
 35  * @param {Object} data
 36  * @param {function} callback
 37  */
 38 CommonDao.prototype.isExist = function(colname, id, callback) {
 39 	var col = colmap[colname];
 40 	if (!col) {
 41 		return callback("no such table colname:["+colname+"]");
 42 	}
 43 
 44 	var data = {_id: id};
 45 	list(colname, {where: data}, function(err, result) {
 46 		callback(err, 0 < result.length);
 47 	});
 48 };

 75 /**
 76  * データベース内の一覧取得
 77  * @param {String} colname コレクション名
 78  * @param {Object} data
 79  * @param {function} callback
 80  */
 81 var list = CommonDao.prototype.list = function(colname, data, callback) {
 82 	var col = colmap[colname];
 83 
 84 	if (!col) {
 85 		return callback("no such table colname:["+colname+"]");
 86 	}
 87 
 88 	//console.log('検索条件', colname, data);
 89 	col
 90 		.find(data.where, data.select, {sort: data.sort, skip: data.skip, limit: data.limit}, function(err, result) {
 91 			//console.log('検索結果', result);
 92 			if (err) {
 93 				console.log(err);
 94 			}
 95 			callback(err, result || []);
 96 		});
 97 };

123 /**
124  * データを更新(insert or update)
125  * @param {String} colname コレクション名
126  * @param {Object} criteria
127  * @param {Object} data
128  * @param {function} callback
129  */
130 CommonDao.prototype.upsert = function(colname, criteria, data, callback) {
131 	var col = colmap[colname];
132 
133 	if (!col) {
134 		return callback("no such table colname:["+colname+"]");
135 	}
136 	if (!data || data._id) {
137 		return callback("valid data:["+data+"]");
138 	}
139 
140 	if (!criteria) {
141 		criteria = {};
142 	}
143 	if (!criteria._id) {
144 		criteria._id = makeObjectId();
145 		data.registTime = Date.now();
146 	}
147 	data.updateTime = Date.now();
148 
149 	console.log('データ登録/更新', colname, criteria, data);
150 	col
151 		.update(criteria, {$set: data}, {upsert: true}, function(err) {
152 
153 			if (err) {
154 				console.log(err);
155 			}
156 			if(!data._id) {
157 				data._id = criteria._id;
158 			};
159 			callback(err, data);
160 		});
161 };
