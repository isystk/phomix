  7 var User = function(connection) {
  8 	this.id = undefined; // ユーザーID
  9 	this.name = undefined; // ユーザー名
 10 	this.connection = connection;
 11 };
 12 
 13 module.exports = User;
 14 
 15 /**
 16  * データ送信
 17  * @param {String} method
 18  * @param {Object} data
 19  */
 20 User.prototype.send = function(method, data) {
 21 	try {
 22 		var json = JSON.stringify(data);
 23 		//this.connection.send(method + ':' + json);
 24 		this.connection.emit('message', method + ':' + json);
 25 		console.log('send', method, json);
 26 	} catch (e) {
 27 		console.log(e);
 28 	}
 29 };
 30 
 31 /**
 32  * データ送信(接続中の全ユーザーに送信する)
 33  * @param {String} method
 34  * @param {Object} data
 35  */
 36 User.prototype.sendAll = function(method, data) {
 37 	try {
 38 		var json = JSON.stringify(data);
 39 		this.connection.broadcast.emit('message', method + ':' + json);
 40 		console.log('sendAll', method, json);
 41 	} catch (e) {
 42 		console.log(e);
 43 	}
 44 };
 45 
 46 /**
 47  * ERRORデータ送信
 48  * @param {Error} err
 49  */
 50 User.prototype.sendError = function(err) {
 51 	this.send('error', {
 52 		name: err.name,
 53 		message: err.message
 54 	});
 55 };
