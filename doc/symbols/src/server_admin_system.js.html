  8 var adminService = require('../service/admin');
  9 var commonDao = require('../dao/common');
 10 
 11 function SystemController() {
 12 }
 13 
 14 module.exports = new SystemController();
 15 
 16 /**
 17  * レベルの一覧を取得
 18  * @param user
 19  * @param data
 20  */
 21 SystemController.prototype.list = function(user, data) {
 22 	adminService.list('system', {selector: {}, page: data.page, sort: {system: 1}}, function(err, result) {
 23 		user.send('system.list', result);
 24 	});
 25 };
 26 
 27 
 28 /**
 29  * レベルの保存
 30  * @param user
 31  * @param data
 32  */
 33 SystemController.prototype.save = function(user, data) {
 34 	var criteria = {};
 35 	if (data._id === '') {
 36 		data._id = null;
 37 	}
 38 	criteria._id = commonDao.ObjectID(data._id);
 39 	delete data._id;
 40 	adminService.save('system', criteria, data, function(err, result) {
 41 		user.send('system.save', result);
 42 	});
 43 };
 44 
 45 
 46 /**
 47  * レベルの削除
 48  * @param user
 49  * @param data
 50  */
 51 SystemController.prototype.remove = function(user, data) {
 52 	var criteria = {};
 53 	criteria._id = commonDao.ObjectID(data._id);
 54 	adminService.remove('system', criteria, function(err, result) {
 55 		user.send('system.remove', result);
 56 	});
 57 };
