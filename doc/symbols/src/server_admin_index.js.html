  6 var applications = require('../applications');
  7 
  8 var Server = function() {
  9 	this.app = undefined;
 10 	this.names = ['admin', 'user', 'system'];
 11 };
 12 
 13 module.exports = new Server();
 14 
 15 /**
 16  * コントローラーを登録する
 17  * @param {Array} names コントローラー名
 18  */
 19 Server.prototype.registController = function() {
 20 	var self = this;
 21 	this.names.forEach(function(name) {
 22 		console.log('Searching ' + name + ' controler');
 23 		var control = require('./'+name);
 24 		for (var method in control) {
 25 			if (method) {
 26 				var ns = name + '.' + method;
 27 				console.log('Add handler [' + ns + ']');
 28 				self.app.handlers[ns] = control[method];
 29 			}
 30 		}
 31 	});
 32 };
 33 
 34 /**
 35  * 管理用レスポンス
 36  * @param user
 37  * @param data
 38  * @return
 39  */
 40 function AdminResponse(user, data) {
 41 	this.id = data._req;
 42 	this.user = user;
 43 	this.connection = user.connection;
 44 }
 45 /**
 46  * 管理ユーザーに返信します
 47  * @param obj
 48  * @return
 49  */
 50 AdminResponse.prototype.reply = function(obj) {
 51 	obj._req = this.id;
 52 	this.user.send('_', obj);
 53 };
 54 /**
 55  * 管理ユーザーにエラーを通達します
 56  * @param err
 57  * @return
 58  */
 59 AdminResponse.prototype.error = function(err) {
 60 	obj._req = this.id;
 61 	user.sendError(err);
 62 };

 65 /**
 66  * コントローラーへ処理を移譲
 67  * @param {Connection} user ユーザー
 68  * @param {String} method メソッド名
 69  * @param {Object} data クライアントからのデータ(JSON)
 70  */
 71 Server.prototype.invoke = function(user, method, data) {
 72 
 73 	try {
 74 		var handler = this.app.handlers[method];
 75 		var response = new AdminResponse(user, data);
 76 		if (handler === undefined) {
 77 			conn.send('error:' + JSON.stringify({message:'存在しないメソッドが呼ばれました[' + method + ']'}));
 78 		} else {
 79 			handler(user, data, response);
 80 		}
 81 	} catch (e) {
 82 		console.log(e);
 83 	}
 84 };

 86 Server.prototype.setup = function(app) {
 87 
 88 	var self = this;
 89 	this.app = app;
 90 
 91 	/**
 92 	 * ユーザーがWebSocketに接続した時の処理
 93 	 */
 94 	var onConnection = function(conn) {
 95 		console.log("connection - start");
 96 
 97 		// ユーザー情報の初期構成
 98 		var User = require('../user');
 99 		var user = new User(conn);
100 		applications.putUser(user);
101 
102 		conn.on('message', function (message) {
103 
104 			console.log('received message', user.connection.id, message);
105 
106 			try {
107 				var midx = message.indexOf(':');
108 				var method = message.substr(0, midx);
109 				var json = message.substr(midx+1);
110 				var data = JSON.parse(json);
111 
112 				// 認証チェック
113 				if (!user.session &&
114 					method != 'admin.login' &&
115 					method != 'admin.loginSession' &&
116 					!user.name
117 				){
118 					user.sendError(new Error('ログインが必要です'));
119 					//adminUser.sendError({
120 					//  name: "name!!",
121 					//  message: "message!!"
122 					//})
123 				} else {
124 					// コントローラメソッド実行
125 					self.invoke(user, method, data);
126 				}
127 			} catch (e) {
128 				console.log(e);
129 			}
130 		});
131 
132 		conn.on('disconnect', function () {
133 			// アプリケーションからユーザーを削除します。
134 			if (user.id) {
135 				applications.removeUser(user);
136 			}
137 			console.log("connection - disconnect");
138 		});
139 	};
140 
141 	// ユーザー接続時のリスナーを設定します。
142 	app.emit('onregister', onConnection);
143 
144 	// コントローラを設定します。
145 	this.registController();
146 
147 };
