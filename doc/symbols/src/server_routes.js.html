  6 
  7 var Routes = function() {};
  8 
  9 module.exports = new Routes();
 10 
 11 /**
 12  * 自動でroutes/フォルダのURLルーティングを行う。
 13  */
 14 Routes.prototype.setup = function(route, app) {
 15 	/**
 16 	 * ルーティングの登録を行う
 17 	 * @param {Object} route 定義オブジェクト
 18 	 * @param {Object} app サービス
 19 	 */
 20 	var register = function(route, app) {
 21 		var routes = {'get': route.gets, 'post': route.posts, 'put': route.puts, 'delete': route.deletes};
 22 		for (var route in routes) {
 23 			for (var uri in routes[route]) {
 24 				var args = routes[route][uri];
 25 				if (args instanceof Array) {
 26 					app[route](uri, args[0], args[1]);
 27 				} else {
 28 					app[route](uri, args);
 29 				}
 30 				console.log('Add Routing ' + route, uri);
 31 			}
 32 		}
 33 	};
 34 
 35 	register(route, app);
 36 
 37 };
