1 /**
  2  * @fileOverview コントローラー
  3  * @name index.js
  4  * @author Yoshitaka Ise <y-ise@ise-web.com>
  5  */
  6 var applications = require('../applications');
  7 
  8 var Server = function() {
  9 	this.app = undefined;
 10 	this.names = ['login', 'loginIdentify', 'loginRegister', 'myDashboard', 'myPost', 'mySettings', 'tag', 'top', 'user'];
 11 };
 12 
 13 module.exports = new Server();
 14 
 15 /**
 16  * コントローラーを登録する
 17  * @param {Array} names コントローラー名
 18  */
 19 Server.prototype.registController = function() {
 20 	var self = this;
 21 	this.names.forEach(function(name) {
 22 		console.log('Searching ' + name + ' controler');
 23 		var control = require('./'+name);
 24 		for (var method in control) {
 25 			if (method) {
 26 				var ns = name + '.' + method;
 27 				console.log('Add handler [' + ns + ']');
 28 				self.app.handlers[ns] = control[method];
 29 			}
 30 		}
 31 	});
 32 };
 33 
 34 /**
 35  * コントローラーへ処理を移譲
 36  * @param {Connection} user ユーザー
 37  * @param {String} method メソッド名
 38  * @param {Object} data クライアントからのデータ(JSON)
 39  */
 40 Server.prototype.invoke = function(user, method, data) {
 41 	try {
 42 		var handler = this.app.handlers[method];
 43 		if (handler === undefined) {
 44 			conn.send('error:' + JSON.stringify({message:'存在しないメソッドが呼ばれました[' + method + ']'}));
 45 		} else {
 46 			handler(user, data);
 47 		}
 48 	} catch (e) {
 49 		console.log(e);
 50 	}
 51 };
 52 
 53 Server.prototype.setup = function(app, socket) {
 54 
 55 	var self = this;
 56 	this.app = app;
 57 
 58 	/**
 59 	 * ユーザーがWebSocketに接続した時の処理
 60 	 */
 61 	var onConnection = function(conn) {
 62 		console.log("connection - start");
 63 
 64 		// ユーザー情報の初期構成
 65 		var User = require('../user');
 66 		var user = new User(conn);
 67 		applications.putUser(user);
 68 
 69 		conn.on('message', function (message) {
 70 
 71 			console.log('received message', user.connection.id, message);
 72 
 73 			try {
 74 				var midx = message.indexOf(':');
 75 				var method = message.substr(0, midx);
 76 				var json = message.substr(midx+1);
 77 				var data = JSON.parse(json);
 78 
 79 				// コントローラメソッド実行
 80 				self.invoke(user, method, data);
 81 			} catch (e) {
 82 				console.log(e);
 83 			}
 84 		});
 85 
 86 		conn.on('disconnect', function () {
 87 			// アプリケーションからユーザーを削除します。
 88 			if (user.id) {
 89 				applications.removeUser(user);
 90 			}
 91 			console.log("connection - disconnect");
 92 		});
 93 	};
 94 	// ユーザー接続時のリスナーを設定します。
 95 	app.emit('onregister', onConnection);
 96 
 97 	// コントローラを設定します。
 98 	this.registController();
 99 
100 };
